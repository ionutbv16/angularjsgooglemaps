Styling above list stations gradient with SCSS Preprocessor

using 'for' + 'if' conditions 

$htmlFont :'Roboto', sans-serif;
$stationBgColor: #eeeeee;
$stationTitleColor: #006699;
$stationTitleSize: 22px;
$stationTitleTopMargin: 20px;
$borderRadius: 6px;
$paddingStations: 2px;

html, body{
  font-family: $htmlFont;
}

$prct-blue:3%;
$blue: #006699;
$white: #ffffff;


@for $i from 0 through 4 {
    .inner_station p:nth-child(#{$i}) {
        background-color: mix($blue, $white, $prct-blue); border-radius: $borderRadius; padding:$paddingStations;
        @if ($i == 1) {
            font-size: $stationTitleSize; margin-top: $stationTitleTopMargin; color: $stationTitleColor;  
        }
        @if ($i == 4) {
            font-style:italic;   
        }
    }
    $prct-blue: $prct-blue + 5%;
}
